001package software.leonov.common.util.function;
002
003import static software.leonov.common.util.function.Exceptions.uncheckedException;
004
005import java.util.Objects;
006import java.util.function.Consumer;
007
008/**
009 * Mirror of the {@link Consumer} interface whose {@code accept(T)} method can throw a checked exception.
010 */
011@FunctionalInterface
012public interface CheckedConsumer<T> {
013
014    /**
015     * Performs this operation on the given argument.
016     *
017     * @param t the input argument
018     * @throws Exception if an error occurs
019     */
020    public void accept(T t) throws Exception;
021
022    /**
023     * Returns a composed consumer that performs, in sequence, this operation followed by the {@code after} operation. If
024     * performing either operation throws an exception, it is relayed to the caller of the composed operation. If performing
025     * {@code this} operation throws an exception, the {@code after} operation will not be performed.
026     *
027     * @param after the operation to perform after this operation
028     * @return a composed consumer that performs in sequence this operation followed by the {@code after} operation
029     */
030    public default CheckedConsumer<T> andThen(final CheckedConsumer<? super T> after) {
031        Objects.requireNonNull(after, "after == null");
032        return t -> {
033            accept(t);
034            after.accept(t);
035        };
036    }
037
038    /**
039     * Returns a {@link Consumer} which delegates to the underlying {@link CheckedConsumer},
040     * {@link Exceptions#uncheckedException(Exception) rethrowing} any checked exceptions as if they were unchecked.
041     * 
042     * @param consumer the underlying checked consumer
043     * @return a {@link Consumer} which delegates to the underlying {@link CheckedConsumer},
044     *         {@link Exceptions#uncheckedException(Exception) rethrowing} any checked exceptions as if they were unchecked
045     */
046    public static <T> Consumer<T> evalUnchecked(final CheckedConsumer<? super T> consumer) {
047        Objects.requireNonNull(consumer, "consumer == null");
048        return t -> {
049            try {
050                consumer.accept(t);
051            } catch (final Exception e) {
052                throw uncheckedException(e);
053            }
054        };
055    }
056
057}
