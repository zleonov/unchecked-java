001package software.leonov.common.util.function;
002
003import static software.leonov.common.util.function.Exceptions.uncheckedException;
004
005import java.util.Objects;
006import java.util.function.DoubleUnaryOperator;
007
008/**
009 * Mirror of the {@link DoubleUnaryOperator} interface whose {@code applyAsDouble(double)} method can throw a checked
010 * exception.
011 */
012@FunctionalInterface
013public interface CheckedDoubleUnaryOperator {
014
015    /**
016     * Applies this operator to the given operand.
017     *
018     * @param operand the the operand
019     * @return the operator result
020     * @throws Exception if an error occurs
021     */
022    public double applyAsDouble(final double operand) throws Exception;
023
024    /**
025     * Returns a composed operator that first applies the {@code before} operator to its input, and then applies this
026     * operator to the result. If evaluation of either operator throws an exception, it is relayed to the caller of the
027     * composed operator.
028     *
029     * @param before the operator to apply before this operator is applied
030     * @return a composed operator that first applies the {@code before} operator and then applies this operator
031     *
032     * @see #andThen(CheckedDoubleUnaryOperator)
033     */
034    default CheckedDoubleUnaryOperator compose(final CheckedDoubleUnaryOperator before) {
035        Objects.requireNonNull(before, "before == null");
036        return v -> applyAsDouble(before.applyAsDouble(v));
037    }
038
039    /**
040     * Returns a composed operator that first applies this operator to its input, and then applies the {@code after}
041     * operator to the result. If evaluation of either operator throws an exception, it is relayed to the caller of the
042     * composed operator.
043     *
044     * @param after the operator to apply after this operator is applied
045     * @return a composed operator that first applies this operator and then applies the {@code after} operator
046     *
047     * @see #compose(CheckedDoubleUnaryOperator)
048     */
049    default CheckedDoubleUnaryOperator andThen(final CheckedDoubleUnaryOperator after) {
050        Objects.requireNonNull(after, "after == null");
051        return t -> after.applyAsDouble(applyAsDouble(t));
052    }
053
054    /**
055     * Returns a {@link DoubleUnaryOperator} which delegates to the underlying {@link CheckedDoubleUnaryOperator},
056     * {@link Exceptions#uncheckedException(Throwable) rethrowing} any checked exceptions as if they were unchecked.
057     * 
058     * @param operator the underlying checked operator
059     * @return a {@link DoubleUnaryOperator} which delegates to the underlying {@link CheckedDoubleUnaryOperator},
060     *         {@link Exceptions#uncheckedException(Throwable) rethrowing} any checked exceptions as if they were unchecked
061     */
062    public static DoubleUnaryOperator evalUnchecked(final CheckedDoubleUnaryOperator operator) {
063        Objects.requireNonNull(operator, "operator == null");
064        return d -> {
065            try {
066                return operator.applyAsDouble(d);
067            } catch (final Exception e) {
068                throw uncheckedException(e);
069            }
070        };
071    }
072
073}
