001package software.leonov.common.util.function;
002
003import static software.leonov.common.util.function.Exceptions.uncheckedException;
004
005import java.util.Objects;
006import java.util.function.ObjLongConsumer;
007
008/**
009 * Mirror of the {@link ObjLongConsumer} interface whose {@code accept(T, long)} method can throw a checked exception.
010 */
011@FunctionalInterface
012public interface CheckedObjLongConsumer<T> {
013
014    /**
015     * Performs this operation on the given arguments.
016     *
017     * @param t     the first argument
018     * @param value the second argument
019     * @throws Exception if an error occurs
020     */
021    public void accept(final T t, final long value) throws Exception;
022
023    /**
024     * Returns an {@link ObjLongConsumer} which delegates to the underlying {@link CheckedObjLongConsumer},
025     * {@link Exceptions#uncheckedException(Exception) rethrowing} any checked exceptions as if they were unchecked.
026     * 
027     * @param consumer the underlying checked consumer
028     * @return an {@link ObjLongConsumer} which delegates to the underlying {@link CheckedObjLongConsumer},
029     *         {@link Exceptions#uncheckedException(Exception) rethrowing} any checked exceptions as if they were unchecked
030     */
031    public static <T> ObjLongConsumer<T> evalUnchecked(final CheckedObjLongConsumer<? super T> consumer) {
032        Objects.requireNonNull(consumer, "consumer == null");
033        return (t, l) -> {
034            try {
035                consumer.accept(t, l);
036            } catch (final Exception e) {
037                throw uncheckedException(e);
038            }
039        };
040    }
041}
