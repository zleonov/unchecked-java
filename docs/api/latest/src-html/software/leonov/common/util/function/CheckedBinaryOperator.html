001package software.leonov.common.util.function;
002
003import static software.leonov.common.util.function.Exceptions.uncheckedException;
004
005import java.util.Objects;
006import java.util.function.BinaryOperator;
007
008/**
009 * Mirror of the {@link BinaryOperator} interface whose {@code apply(T, T)} method can throw a checked exception.
010 */
011@FunctionalInterface
012public interface CheckedBinaryOperator<T> extends CheckedBiFunction<T, T, T> {
013
014    /**
015     * Returns a {@link CheckedBinaryOperator} which returns the lesser of two elements according to the specified
016     * {@code CheckedComparator}.
017     *
018     * @param comparator a {@code CheckedComparator} for comparing the two values
019     * @return a {@code CheckedBinaryOperator} which returns the lesser of its operands, according to the supplied
020     *         {@code CheckedComparator}
021     */
022    public static <T> CheckedBinaryOperator<T> minBy(final CheckedComparator<? super T> comparator) {
023        Objects.requireNonNull(comparator, "comparator == null");
024        return (a, b) -> comparator.compare(a, b) <= 0 ? a : b;
025    }
026
027    /**
028     * Returns a {@link CheckedBinaryOperator} which returns the greater of two elements according to the specified
029     * {@code Comparator}.
030     *
031     * @param comparator a {@code CheckedComparator} for comparing the two values
032     * @return a {@code CheckedBinaryOperator} which returns the greater of its operands, according to the supplied
033     *         {@code CheckedComparator}
034     */
035    public static <T> CheckedBinaryOperator<T> maxBy(final CheckedComparator<? super T> comparator) {
036        Objects.requireNonNull(comparator, "comparator == null");
037        return (a, b) -> comparator.compare(a, b) >= 0 ? a : b;
038    }
039
040    /**
041     * Returns a {@link BinaryOperator} which delegates to the underlying {@link CheckedBinaryOperator},
042     * {@link Exceptions#uncheckedException(Throwable) rethrowing} any checked exceptions as if they were unchecked.
043     * 
044     * @param operator the underlying checked operator
045     * @return a {@link BinaryOperator} which delegates to the underlying {@link CheckedBinaryOperator},
046     *         {@link Exceptions#uncheckedException(Throwable) rethrowing} any checked exceptions as if they were unchecked
047     */
048    public static <T> BinaryOperator<T> evalUnchecked(final CheckedBinaryOperator<T> operator) {
049        Objects.requireNonNull(operator, "operator == null");
050        return (t, u) -> {
051            try {
052                return operator.apply(t, u);
053            } catch (final Exception e) {
054                throw uncheckedException(e);
055            }
056        };
057    }
058
059}
