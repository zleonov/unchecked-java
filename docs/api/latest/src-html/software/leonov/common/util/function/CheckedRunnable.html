001package software.leonov.common.util.function;
002
003import static software.leonov.common.util.function.Exceptions.uncheckedException;
004
005import java.util.Objects;
006
007/**
008 * Mirror of the {@link Runnable} interface whose {@code run()} method can throw a checked exception.
009 */
010@FunctionalInterface
011public interface CheckedRunnable {
012
013    /**
014     * The general contract of the method {@code run()} is that it may take any action whatsoever.
015     * 
016     * @throws Exception if an error occurs
017     */
018    public void run() throws Exception;
019
020    /**
021     * Returns a {@link Runnable} which delegates to the underlying {@link CheckedRunnable},
022     * {@link Exceptions#uncheckedException(Exception) rethrowing} any checked exceptions as if they were unchecked.
023     * 
024     * @param runnable the underlying checked runnable
025     * @return a {@link Runnable} which delegates to the underlying {@link CheckedRunnable},
026     *         {@link Exceptions#uncheckedException(Exception) rethrowing} any checked exceptions as if they were unchecked
027     */
028    public static Runnable evalUnchecked(final CheckedRunnable runnable) {
029        Objects.requireNonNull(runnable, "runnable == null");
030        return () -> {
031            try {
032                runnable.run();
033            } catch (final Exception e) {
034                throw uncheckedException(e);
035            }
036        };
037    }
038
039}
