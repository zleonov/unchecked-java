001package software.leonov.common.util.function;
002
003import static software.leonov.common.util.function.Exceptions.uncheckedException;
004
005import java.util.Objects;
006import java.util.function.Function;
007
008/**
009 * Mirror of the {@link Function} interface whose {@code apply(T)} method can throw a checked exception.
010 */
011@FunctionalInterface
012public interface CheckedFunction<T, R> {
013
014    /**
015     * Applies this function to the given argument.
016     *
017     * @param t the function argument
018     * @return the function result
019     * @throws Exception if an error occurs
020     */
021    public R apply(final T t) throws Exception;
022
023    /**
024     * Returns a composed function that first applies the {@code before} function to its input, and then applies this
025     * function to the result. If evaluation of either function throws an exception, it is relayed to the caller of the
026     * composed function.
027     *
028     * @param before the function to apply before this function is applied
029     * @return a composed function that first applies the {@code before} function and then applies this function
030     *
031     * @see #andThen(CheckedFunction)
032     */
033    public default <V> CheckedFunction<V, R> compose(final CheckedFunction<? super V, ? extends T> before) {
034        Objects.requireNonNull(before, "before == null");
035        return v -> apply(before.apply(v));
036    }
037
038    /**
039     * Returns a composed function that first applies this function to its input, and then applies the {@code after}
040     * function to the result. If evaluation of either function throws an exception, it is relayed to the caller of the
041     * composed function.
042     *
043     * @param after the function to apply after this function is applied
044     * @return a composed function that first applies this function and then applies the {@code after} function
045     *
046     * @see #compose(CheckedFunction)
047     */
048    public default <V> CheckedFunction<T, V> andThen(final CheckedFunction<? super R, ? extends V> after) {
049        Objects.requireNonNull(after, "after == null");
050        return t -> after.apply(apply(t));
051    }
052
053    // Any point in having CheckedFunction.identity() method corresponding to Function.identity()?
054
055    /**
056     * Returns a {@link Function} which delegates to the underlying {@link CheckedFunction},
057     * {@link Exceptions#uncheckedException(Exception) rethrowing} any checked exceptions as if they were unchecked.
058     * 
059     * @param function the underlying checked function
060     * @return a {@link Function} which delegates to the underlying {@link CheckedFunction},
061     *         {@link Exceptions#uncheckedException(Exception) rethrowing} any checked exceptions as if they were unchecked
062     */
063    public static <T, R> Function<T, R> evalUnchecked(final CheckedFunction<? super T, ? extends R> function) {
064        Objects.requireNonNull(function, "function == null");
065        return t -> {
066            try {
067                return function.apply(t);
068            } catch (final Exception e) {
069                throw uncheckedException(e);
070            }
071        };
072    }
073
074}
