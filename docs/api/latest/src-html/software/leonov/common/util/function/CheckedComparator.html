001package software.leonov.common.util.function;
002
003import static software.leonov.common.util.function.Exceptions.uncheckedException;
004
005import java.io.Serializable;
006import java.util.Comparator;
007import java.util.Objects;
008
009/**
010 * Mirror of the {@link Comparator} interface whose {@code compare(T, T)} method can throw a checked exception.
011 */
012@FunctionalInterface
013public interface CheckedComparator<T> {
014
015    /**
016     * Compares its two arguments for order.
017     *
018     * @param left  the first argument
019     * @param right the second argument
020     * @return a negative integer, zero, or a positive integer as the first argument is less than, equal to, or greater than
021     *         the second
022     * @throws Exception if an error occurs
023     */
024    public int compare(final T left, final T right) throws Exception;
025
026    /**
027     * Returns a comparator that applies the specified function to each element of {@code type T} and compares the results
028     * according to their <i>natural ordering</i>.
029     * <p>
030     * The returned comparator is serializable if the specified function is also serializable.
031     *
032     * @param <T>      the type of element to be compared
033     * @param <U>      the type of the result returned by the function
034     * @param function the function to apply
035     * @return a comparator that applies the specified function to each element of {@code type T} and compares the results
036     *         according to their <i>natural ordering</i>
037     */
038    @SuppressWarnings("unchecked")
039    public static <T, U extends Comparable<? super U>> CheckedComparator<T> comparing(final CheckedFunction<? super T, ? extends U> function) {
040        Objects.requireNonNull(function, "function == null");
041        return (CheckedComparator<T> & Serializable) (left, right) -> function.apply(left).compareTo(function.apply(right));
042    }
043
044    /**
045     * Returns a comparator that uses the specified comparator to compare the results of applying the specified function to
046     * each element of {@code type T}.
047     * <p>
048     * The returned comparator is serializable if both the function and the specified comparator are serializable.
049     *
050     * @param <T>        the type of element to be compared
051     * @param <U>        the type of the result returned by the function
052     * @param function   the function to apply
053     * @param comparator the comparator to use to compare elements of {@code type U}
054     * @return a comparator that uses the specified comparator to compare the results of applying the specified function to
055     *         each element of {@code type T}
056     */
057    @SuppressWarnings("unchecked")
058    public static <T, U> CheckedComparator<T> comparing(final CheckedFunction<? super T, ? extends U> function, CheckedComparator<? super U> comparator) {
059        Objects.requireNonNull(function, "function == null");
060        Objects.requireNonNull(comparator, "comparator == null");
061        return (CheckedComparator<T> & Serializable) (left, right) -> comparator.compare(function.apply(left), function.apply(right));
062    }
063
064    /**
065     * Returns a comparator that applies the specified function to each element of {@code type T} and compares the
066     * {@code double} results according to their <i>natural ordering</i>.
067     * <p>
068     * The returned comparator is serializable if the specified function is also serializable.
069     *
070     * @param <T>      the type of element to be compared
071     * @param function the function to apply
072     * @return a comparator that applies the specified function to each element of {@code type T} and compares the
073     *         {@code double} results according to their <i>natural ordering</i>
074     */
075    @SuppressWarnings("unchecked")
076    public static <T> CheckedComparator<T> comparingDouble(final CheckedToDoubleFunction<? super T> function) {
077        Objects.requireNonNull(function, "function == null");
078        return (CheckedComparator<T> & Serializable) (left, right) -> Double.compare(function.applyAsDouble(left), function.applyAsDouble(right));
079    }
080
081    /**
082     * Returns a comparator that applies the specified function to each element of {@code type T} and compares the
083     * {@code int} results according to their <i>natural ordering</i>.
084     * <p>
085     * The returned comparator is serializable if the specified function is also serializable.
086     *
087     * @param <T>      the type of element to be compared
088     * @param function the function to apply
089     * @return a comparator that applies the specified function to each element of {@code type T} and compares the
090     *         {@code int} results according to their <i>natural ordering</i>
091     */
092    @SuppressWarnings("unchecked")
093    public static <T> CheckedComparator<T> comparingInt(final CheckedToIntFunction<? super T> function) {
094        Objects.requireNonNull(function, "function == null");
095        return (CheckedComparator<T> & Serializable) (left, right) -> Integer.compare(function.applyAsInt(left), function.applyAsInt(right));
096    }
097
098    /**
099     * Returns a comparator that applies the specified function to each element of {@code type T} and compares the
100     * {@code long} results according to their <i>natural ordering</i>.
101     * <p>
102     * The returned comparator is serializable if the specified function is also serializable.
103     *
104     * @param <T>      the type of element to be compared
105     * @param function the function to apply
106     * @return a comparator that applies the specified function to each element of {@code type T} and compares the
107     *         {@code long} results according to their <i>natural ordering</i>
108     */
109    @SuppressWarnings("unchecked")
110    public static <T> CheckedComparator<T> comparingLong(final CheckedToLongFunction<? super T> function) {
111        Objects.requireNonNull(function, "function == null");
112        return (CheckedComparator<T> & Serializable) (left, right) -> Long.compare(function.applyAsLong(left), function.applyAsLong(right));
113    }
114
115    /**
116     * Returns a comparator that compares values according to their <i>natural ordering</i>.
117     * <p>
118     * The returned comparator is serializable and does not accept {@code null} values.
119     *
120     * @param <T> the type of element to be compared
121     * @return a comparator that compares values according to their <i>natural ordering</i>
122     */
123    @SuppressWarnings("unchecked")
124    public static <T extends Comparable<? super T>> CheckedComparator<T> naturalOrder() {
125        return (CheckedComparator<T>) NaturalOrdering.getInstance();
126    }
127
128    /**
129     * Returns a comparator that considers {@code null} values to be less than non-{@code null} values. If both are
130     * non-{@code null}, the specified comparator is used to determine the order. If the specified comparator is
131     * {@code null}, then the returned comparator considers all non-{@code null} values to be equal.
132     * <p>
133     * The returned comparator is serializable if the specified comparator is serializable.
134     *
135     * @param <T>        the type of the elements to be compared
136     * @param comparator the comparator to use for comparing non-{@code null} values
137     * @return a comparator that considers {@code null} values to be less than non-{@code null} values
138     */
139    public static <T> CheckedComparator<T> nullsFirst(final CheckedComparator<? super T> comparator) {
140        return Nulls.first(comparator);
141    }
142
143    /**
144     * Returns a comparator that considers {@code null} values to be greater than non-{@code null} values. If both are
145     * non-{@code null}, the specified comparator is used to determine the order. If the specified comparator is
146     * {@code null}, then the returned comparator considers all non-{@code null} values to be equal.
147     * <p>
148     * The returned comparator is serializable if the specified comparator is serializable.
149     *
150     * @param <T>        the type of the elements to be compared
151     * @param comparator the comparator to use for comparing non-{@code null} values
152     * @return a comparator that considers {@code null} values to be greater than non-{@code null} values
153     */
154    public static <T> CheckedComparator<T> nullsLast(final CheckedComparator<? super T> comparator) {
155        return Nulls.last(comparator);
156    }
157
158    /**
159     * Returns a comparator that compares values according to the reverse of their <i>natural ordering</i>.
160     * <p>
161     * The returned comparator is serializable and does not accept {@code null} values.
162     *
163     * @param <T> the {@link Comparable} type of element to be compared
164     * @return a comparator that compares values according to the reverse of their <i>natural ordering</i>
165     */
166    @SuppressWarnings("unchecked")
167    public static <T extends Comparable<? super T>> CheckedComparator<T> reverseOrder() {
168        return (CheckedComparator<T>) ReversedNaturalOrdering.getInstance();
169    }
170
171    /**
172     * Returns a comparator that has the reverse ordering of this comparator.
173     *
174     * @return a comparator that has the reverse ordering of this comparator
175     */
176    default CheckedComparator<T> reversed() {
177        return new Reversed<T>(this);
178    }
179
180    /**
181     * Returns a composed comparator which uses {@code this} comparator, followed by the {@code other} comparator if this
182     * comparator returns a zero result.
183     * <p>
184     * The returned comparator is serializable if the specified comparator is also serializable.
185     *
186     * @param other the other comparator to be used when {@code this} comparator determines that two objects are equal
187     * @return a composed comparator which uses {@code this} comparator, followed by the {@code other} comparator if this
188     *         comparator returns a zero result
189     */
190    @SuppressWarnings("unchecked")
191    default CheckedComparator<T> thenComparing(final CheckedComparator<? super T> other) {
192        Objects.requireNonNull(other, "other == null");
193        return (CheckedComparator<T> & Serializable) (left, right) -> {
194            int res = compare(left, right);
195            return (res != 0) ? res : other.compare(left, right);
196        };
197    }
198
199    /**
200     * Returns a composed comparator which uses {@code this} comparator first, and if the result is zero, applies the
201     * specified function before comparing the results according to their <i>natural ordering</i>.
202     * <p>
203     * The returned comparator is serializable if the specified comparator is also serializable.
204     *
205     * @param function the function to apply
206     * @return a composed comparator which uses {@code this} comparator first, and if the result is zero, applies the
207     *         specified function before comparing the results according to their <i>natural ordering</i>
208     */
209    default <U extends Comparable<? super U>> CheckedComparator<T> thenComparing(CheckedFunction<? super T, ? extends U> function) {
210        return thenComparing(comparing(function));
211    }
212
213    /**
214     * Returns a composed comparator which uses {@code this} comparator first, and if the result is zero, applies the
215     * specified function before using the {@code other} comparator.
216     * <p>
217     * The returned comparator is serializable if the specified comparator is also serializable.
218     * 
219     * @param function the function to apply
220     * @param other    the comparator to use after applying the function
221     * @return a composed comparator which uses {@code this} comparator first, and if the result is zero, applies the
222     *         specified function before using the {@code other} comparator
223     */
224    default <U> CheckedComparator<T> thenComparing(final CheckedFunction<? super T, ? extends U> function, final CheckedComparator<? super U> other) {
225        return thenComparing(comparing(function, other));
226    }
227
228    /**
229     * Returns a composed comparator which uses {@code this} comparator first, and if the result is zero, applies the
230     * specified function before comparing the {@code double} results according to their <i>natural ordering</i>.
231     * <p>
232     * The returned comparator is serializable if the specified comparator is also serializable.
233     *
234     * @param function the function to apply
235     * @return a composed comparator which uses {@code this} comparator first, and if the result is zero, applies the
236     *         specified function before comparing the {@code double} results according to their <i>natural ordering</i>
237     */
238    default CheckedComparator<T> thenComparingDouble(CheckedToDoubleFunction<? super T> function) {
239        return thenComparing(comparingDouble(function));
240    }
241
242    /**
243     * Returns a composed comparator which uses {@code this} comparator first, and if the result is zero, applies the
244     * specified function before comparing the {@code int} results according to their <i>natural ordering</i>.
245     * <p>
246     * The returned comparator is serializable if the specified comparator is also serializable.
247     * 
248     * @param function the function to apply
249     * @return a composed comparator which uses {@code this} comparator first, and if the result is zero, applies the
250     *         specified function before comparing the {@code int} results according to their <i>natural ordering</i>
251     */
252    default CheckedComparator<T> thenComparingInt(final CheckedToIntFunction<? super T> function) {
253        return thenComparing(comparingInt(function));
254    }
255
256    /**
257     * Returns a composed comparator which uses {@code this} comparator first, and if the result is zero, applies the
258     * specified function before comparing the {@code long} results according to their <i>natural ordering</i>.
259     * <p>
260     * The returned comparator is serializable if the specified comparator is also serializable.
261     * 
262     * @param function the function to apply
263     * @return a composed comparator which uses {@code this} comparator first, and if the result is zero, applies the
264     *         specified function before comparing the {@code long} results according to their <i>natural ordering</i>
265     */
266    default CheckedComparator<T> thenComparingLong(final CheckedToLongFunction<? super T> function) {
267        return thenComparing(comparingLong(function));
268    }
269
270    /**
271     * Returns a {@link Comparator} which delegates to the underlying {@link CheckedComparator},
272     * {@link Exceptions#uncheckedException(Exception) rethrowing} any checked exceptions as if they were unchecked.
273     * 
274     * @param comparator the specified checked comparator
275     * @return a {@link Comparator} which delegates to the underlying {@link CheckedComparator},
276     *         {@link Exceptions#uncheckedException(Exception) rethrowing} any checked exceptions as if they were unchecked
277     */
278    public static <T> Comparator<T> evalUnchecked(final CheckedComparator<? super T> comparator) {
279        Objects.requireNonNull(comparator, "comparator == null");
280        return (left, right) -> {
281            try {
282                return comparator.compare(left, right);
283            } catch (final Exception e) {
284                throw uncheckedException(e);
285            }
286        };
287    }
