001package software.leonov.common.util.function;
002
003import static software.leonov.common.util.function.Exceptions.uncheckedException;
004
005import java.util.Objects;
006import java.util.function.LongBinaryOperator;
007
008/**
009 * Mirror of the {@link LongBinaryOperator} interface whose {@code applyAsLong(long, long)} method can throw a
010 * checked exception.
011 */
012@FunctionalInterface
013public interface CheckedLongBinaryOperator {
014
015    /**
016     * Applies this operator to the given operands.
017     * 
018     * @param left  the first operand
019     * @param right the second operand
020     * @return the operator result
021     * @throws Exception if an error occurs
022     */
023    public long applyAsLong(long left, long right) throws Exception;
024
025    /**
026     * Returns a {@link LongBinaryOperator} which delegates to the underlying {@link CheckedLongBinaryOperator},
027     * {@link Exceptions#uncheckedException(Throwable) rethrowing} any checked exceptions as if they were unchecked.
028     * 
029     * @param operator the underlying checked operator
030     * @return a {@link LongBinaryOperator} which delegates to the underlying {@link CheckedLongBinaryOperator},
031     *         {@link Exceptions#uncheckedException(Throwable) rethrowing} any checked exceptions as if they were unchecked
032     */
033    public static LongBinaryOperator evalUnchecked(final CheckedLongBinaryOperator operator) {
034        Objects.requireNonNull(operator, "operator == null");
035        return (l, r) -> {
036            try {
037                return operator.applyAsLong(l, r);
038            } catch (final Exception e) {
039                throw uncheckedException(e);
040            }
041        };
042    }
043
044}
