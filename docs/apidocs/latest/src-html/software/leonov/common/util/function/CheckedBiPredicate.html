001package software.leonov.common.util.function;
002
003import static software.leonov.common.util.function.Exceptions.uncheckedException;
004
005import java.util.Objects;
006import java.util.function.BiPredicate;
007
008/**
009 * Mirror of the {@link BiPredicate} interface whose {@code test(T, U)} method can throw a checked exception.
010 */
011@FunctionalInterface
012public interface CheckedBiPredicate<T, U> {
013
014    /**
015     * Evaluates this predicate on the given arguments.
016     *
017     * @param t the first argument
018     * @param u the second argument
019     * @return {@code true} if the input arguments match the predicate, otherwise {@code false}
020     * @throws Exception if an error occurs
021     */
022    public boolean test(T t, U u) throws Exception;
023
024    /**
025     * Returns a composed predicate that represents a short-circuiting logical AND of this predicate and another. When
026     * evaluating the composed predicate, if this predicate is {@code false}, then the {@code other} predicate is not
027     * evaluated.
028     * <p>
029     * Any exceptions thrown during evaluation of either predicate are relayed to the caller; if evaluation of this
030     * predicate throws an exception, the {@code other} predicate will not be evaluated.
031     *
032     * @param other a predicate that will be logically-ANDed with this predicate
033     * @return a composed predicate that represents the short-circuiting logical AND of this predicate and the {@code other}
034     *         predicate
035     */
036    public default CheckedBiPredicate<T, U> and(final CheckedBiPredicate<? super T, ? super U> other) {
037        Objects.requireNonNull(other, "other == null");
038        return (t, u) -> test(t, u) && other.test(t, u);
039    }
040
041    /**
042     * Returns a predicate that represents the logical negation of this predicate.
043     *
044     * @return a predicate that represents the logical negation of this predicate
045     */
046    public default CheckedBiPredicate<T, U> negate() {
047        return (t, u) -> !test(t, u);
048    }
049
050    /**
051     * Returns a composed predicate that represents a short-circuiting logical OR of this predicate and another. When
052     * evaluating the composed predicate, if this predicate is {@code true}, then the {@code other} predicate is not
053     * evaluated.
054     * <p>
055     * Any exceptions thrown during evaluation of either predicate are relayed to the caller; if evaluation of this
056     * predicate throws an exception, the {@code other} predicate will not be evaluated.
057     *
058     * @param other a predicate that will be logically-ORed with this predicate
059     * @return a composed predicate that represents the short-circuiting logical OR of this predicate and the {@code other}
060     *         predicate
061     */
062    public default CheckedBiPredicate<T, U> or(final CheckedBiPredicate<? super T, ? super U> other) {
063        Objects.requireNonNull(other, "other == null");
064        return (t, u) -> test(t, u) || other.test(t, u);
065    }
066
067    /**
068     * Returns a {@link BiPredicate} which delegates to the underlying {@link CheckedBiPredicate},
069     * {@link Exceptions#uncheckedException(Throwable) rethrowing} any checked exceptions as if they were unchecked.
070     * 
071     * @param predicate the underlying checked predicate
072     * @return a {@link BiPredicate} which delegates to the underlying {@link CheckedBiPredicate},
073     *         {@link Exceptions#uncheckedException(Throwable) rethrowing} any checked exceptions as if they were unchecked
074     */
075    public static <T, U> BiPredicate<T, U> evalUnchecked(final CheckedBiPredicate<? super T, ? super U> predicate) {
076        Objects.requireNonNull(predicate, "predicate == null");
077        return (t, u) -> {
078            try {
079                return predicate.test(t, u);
080            } catch (final Exception e) {
081                throw uncheckedException(e);
082            }
083        };
084    }
085
086}
